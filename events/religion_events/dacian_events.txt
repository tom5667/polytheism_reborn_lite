#Spend money to build a small local shrine 
dacian.009 = {
	type = character_event
	title = dacian.009.t
	desc = dacian.009.desc
	theme = learning
	override_background = { reference= pr_generic_temple }
	immediate = {
		hidden_effect = {
			create_character = {
				employer = root
				random_traits = yes
				dynasty = generate	
				save_scope_as = random_character
				stewardship = 26
			}
				cp:councillor_court_chaplain = {
					save_scope_as = realm_priest
				}
		}
	}
	right_portrait = {
		character = root
		animation = personality_content
	}
	left_portrait = {
		character = scope:realm_priest
	}
	#pay for everything
	option = {
		name = dacian.009.a
		add_character_modifier = {
			modifier = built_local_shrine
			years = 5
		}	
		add_trait = zealous
		remove_short_term_gold = 200
		scope:realm_priest = {
			add_opinion = 20
		}
	}
	#Pay only for the materials
	option = {
		name = dacian.009.b	
		remove_short_term_gold = 50
		scope:realm_priest = {
			add_opinion = 10
		}
	}
	#Don't help them and wish them well 
	option = {
		name = dacian.009.c
		scope:realm_priest = {
			add_opinion = -10
		}
	}
}

dacian.010={
	type = character_event
	title = dacian.010.t
	desc = dacian.010.desc
	theme = learning
	#override_background = { refference = bp1_hills }
	override_background = { reference = wilderness_mountains }
	immediate = {
		random_dummy_gender_soldier_effect = { SCOPE_NAME = sacrifice }
		hidden_effect = {
			create_character = {
				age = {25 40 }
				random_traits = yes
				prowess = {1 10}
				martial = {9 15}
				dynasty = none
				gender = male
				faith = Root.faith
				culture = Root.culture
				location = Root.title_province
				save_scope_as = sacrifice
			}
			# Save them in the designated county.
			$LOCATION$ = {
				set_variable = {
					name = fearless_warrior
					value = scope:sacrifice
					years = fp1_stele_duration_value
				}
			}
			cp:councillor_court_chaplain = {
				save_scope_as = realm_priest
			}
		}
	}

	right_portrait = {
		character = root
		animation = personality_content
	}
	left_portrait = {
		character = scope:sacrifice
		animation = celebrate_sword
	}

	#Choose the soldier
	option = {
		name = dacian.010.a
		trigger_event = {
		 days = 5
	     id = dacian.011
		}
	}
	#Stop the sacrifice
	option = {
		name = dacian.010.b	
		add_character_modifier = {
			modifier = sacrifice_refused 
			
			years = 1
		}
	}
}
