							modifier = volva_good_future

						}

						type = event_toast_effect_good

						add_character_modifier = {
							modifier = volva_bad_future
							years = 2
						}
					}
				}
			}
		}	
	}

	type = character_event

		character = root
		animation = personality_rational
	}
	
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = patient
		}
		modifier = {
			add = -0.2
			has_trait = impatient
		}
		modifier = {
			add = 0.2
			has_trait = shy
		}
		modifier = {
			add = -0.2
			has_trait = gregarious
		}
		modifier = {
			add = 0.2
			has_trait = cynical
		}
		modifier = {
			add = -0.2
			has_trait = zealous
		}
	}
	
	option = { # Romans bad
		name = pr_germanic.003.a
		add_piety = 20
		stress_impact = {
			forgiving = miniscule_stress_impact_gain
			humble = miniscule_stress_impact_gain
		}
	}	
	option = { # Romans and Germans can learn from each other
		name = pr_germanic.003.b
		add_piety = 20
		stress_impact = {
			zealous = miniscule_stress_impact_gain
			arrogant = miniscule_stress_impact_gain
		}
	}
}

pr_germanic.005 = {
	type = character_event
	title = pr_germanic.005.t
	desc = pr_germanic.005.desc
	theme = faith
	override_background = { reference= bedchamber }
	left_portrait = {
		character = root
		#Needs to be changed 
		animation = shock
	}
	weight_multiplier = {
		base = 1
		modifier = {
			add = 0.2
			has_trait = zealous
		}
		modifier = {
			add = -0.2
			has_trait = cynical
		}
		modifier = {
			add = 0.2
			has_trait = brave
		}
		modifier = {
			add = -0.2
			has_trait = craven
		}
		modifier = {
			add = 0.2
			has_trait = ambitious
		}
		modifier = {
			add = -0.2
			has_trait = content
		}
	}
	option = { # Just a bad dream
		name = pr_germanic.005.a
		if = {
			limit = {
				OR = {
					has_trait = zealous
					has_trait = cynical
				}
			}
			stress_impact = {
				zealous = medium_stress_impact_gain
				cynical = miniscule_stress_impact_gain
		   }
		}
		else = {
			add_stress = minor_stress_impact_gain
		}

	}	
}
