save_court_augur_as_effect = {
	hidden_effect = {
		if = { 	
			limit = { exists = court_owner }
			court_owner = {
				random_court_position_holder = {
					type = pr_court_augur_court_position
					limit = {
						is_physically_able = yes
					}
					save_scope_as = $SCOPE_NAME$
				}
			}
		}
	}
}
