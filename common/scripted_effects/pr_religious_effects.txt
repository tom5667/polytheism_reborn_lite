pr_rice_culture_tradition_effect  = {
	if = {
		limit = {
			exits = culture:kemetic
		}
		culture:kemetic = {
			remove_culture_tradition = tradition_dryland_dwellers
			add_culture_tradition = tradition_RICE_children_of_the_nile
			remove_variable = tradition_cooldown
		}
		culture:kemetic = {
			if = {
				limit = { has_cultural_tradition = tradition_dryland_dwellers }
				remove_culture_tradition = tradition_dryland_dwellers
			}
			add_culture_tradition = tradition_RICE_children_of_the_nile
			remove_variable = tradition_cooldown
		}

	}
	if = {
		limit = {
			exists = culture:pr_graeco_egyptian
		}
		culture:pr_graeco_egyptian= {
			remove_culture_tradition = tradition_dryland_dwellers
			add_culture_tradition = tradition_RICE_children_of_the_nile
			remove_variable = tradition_cooldown
		}
		culture:pr_graeco_egyptian = {
			if = {
				limit = { has_cultural_tradition = tradition_legalistic }
				remove_culture_tradition = tradition_legalistic
			}
			add_culture_tradition = tradition_RICE_children_of_the_nile
			remove_variable = tradition_cooldown
		}
		set_global_variable = {
			name = RICE_tradition_is_added
			value = yes
		}
	}
}

#Updated scripted effect that normaly destroys the papacy now the patriarchate
destroy_patriarchate_scripted_effect = {
	save_scope_as = scoped_destroyer
	if = {
		limit = { exists = title:k_orthodox.holder }
		title:k_orthodox.holder = { save_scope_as = scoped_pope }
	}
	add_piety = massive_piety_gain
	root.faith = {
		change_fervor = {
			value = 50
			desc = fervor_gain_papacy_destroyed
		}
	}

	# Destroy the papacy title if it exists.
	if = {
		limit = {
			exists = title:k_orthodox
		}
		destroy_title = title:k_orthodox
	}

	# If the Pope exists, they will suffer some kind of humiliating or tragic fate.
	if = {
		limit = {
			exists = scope:scoped_pope
		}
		if = {
			limit = {
				faith = {
					has_doctrine_parameter = human_sacrifice_active
				}
			}
			scope:scoped_pope = {
				death = {
					death_reason = death_sacrificed_to_gods
					killer = root
				}
			}
		}
		else_if = { 
			limit = {
				OR = {
					faith.religion = faith:hellenic_pagan.religion
					religion = religion:roman_religion
				}
			}
			scope:scoped_pope = {
				death = {
					death_reason = death_crucified
					killer = root
				}
			}
		}
		else = {
			scope:scoped_patriarch = { set_character_faith = root.faith }
		}

		if = {
			limit = { scope:scoped_pope.gold > 0 }
			scope:scoped_pope = {
				hidden_effect = { remove_short_term_gold = scope:scoped_pope.gold }
			}
		}
	}

	# Orthodox is now sad.
	faith:orthodox = {
		change_fervor = {
			value = -100
			desc = fervor_loss_patriarchate_destroyed
		}
		if = {
			limit = { has_doctrine = special_doctrine_ecumenical_christian }
			remove_doctrine = special_doctrine_ecumenical_christian
		}
		if = {
			limit = { has_doctrine = doctrine_spiritual_head }
			remove_doctrine = doctrine_spiritual_head
			hidden_effect = { add_doctrine = doctrine_no_head }
		}
	}
}
